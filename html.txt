function saveProfitRow(index, btn) {
  const row = btn.closest("tr");
  const strategy = row.querySelector('select[name="strategy"]').value;
  const date = row.querySelector('input[name="date"]').value;
  const capital = row.querySelector('input[name="capital"]').value;

    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({

    })
  })
  .then(res => res.json())
  .then(data => {
    if (data.success) {
      // 성공 시 다시 보기 모드로 전환
      row.querySelector(".strategy-cell").innerText = strategy;
      row.querySelector(".date-cell").innerText = date;

      row.querySelector(".profit-cell").innerText = `$${parseFloat(profit).toFixed(2)}`;
      const rate = parseFloat(data.profit_rate);
      row.querySelector(".rate-cell").innerText = isNaN(rate) ? '-' : `${rate.toFixed(2)}%`;

  });
}

// ✅ 수익률 계산 함수 (입력값 변경 시 자동 호출)
function updateProfitSummary() {
  const rows = Array.from(document.querySelectorAll("#profitLogBody tr.profit-row"));
  const table = document.getElementById("profitLogBody");

  // 기존 소계/합계 제거
  document.querySelectorAll(".subtotal-row, .total-row").forEach(e => e.remove());

  let currentMonth = null;

  // 월 누적 값
  let monthlyProfit = 0;
  let monthlyFirstCapital = null;   // ★ 그 달의 첫 행 시드

  // 전체(합계) 누적은 '소계'를 바탕으로 계산
  let subtotalProfitSum = 0;
  let subtotalCapitalSum = 0;       // ★ 각 달의 '첫 시드' 합
  let subtotalCount = 0;

  for (let i = 0; i < rows.length; i++) {
    const row = rows[i];

    // 1. 셀렉터로 엘리먼트 먼저 찾기
    const dateCell    = row.querySelector(".date-cell");
    const capitalCell = row.querySelector(".capital-cell");
    const profitCell  = row.querySelector(".profit-cell");
    const rateCell    = row.querySelector(".rate-cell");

    // 2. 사파리 호환성을 위해 innerText 대신 textContent 사용 및 전처리
    // replace(/[^\d.-]/g, '') : 숫자, 소수점(.), 마이너스(-)만 남기고 제거 ($ 등 특수문자 제거)
    const rawDate    = dateCell ? dateCell.textContent.trim() : "";
    const rawCapital = capitalCell ? capitalCell.textContent.trim().replace(/[^\d.]/g, '') : "";
    const rawProfit  = profitCell ? profitCell.textContent.trim().replace(/[^\d.-]/g, '') : "";

    // 3. 숫자 파싱
    const capital = parseFloat(rawCapital);
    const profit  = parseFloat(rawProfit);
    const month   = rawDate ? rawDate.slice(0, 7) : ""; // 예: "2025-12"

    if (!isNaN(capital) && capital > 0) {
      const rate = (profit / capital * 100).toFixed(2);
      rateCell.innerText = `${rate}%`;
      rateCell.style.color = (parseFloat(rate) < 0) ? '#3498db' : '#ff7f7f';
    } else {
      rateCell.innerText = "-";
      rateCell.style.color = '';
    }
    if (profit > 0) {
      profitCell.style.color = '#ff7f7f';
    } else if (profit < 0) {
      profitCell.style.color = '#3498db';
    } else {
      profitCell.style.color = '';
    }

    // 월 경계 초기화
    if (currentMonth === null) currentMonth = month;

    // ★ 새 달 진입 시, 해당 달의 '첫 시드' 세팅
    if (month !== currentMonth) {
      // (이 블록 이전에 직전 달 소계를 만들어야 하므로, 로직 순서를 아래 '달 종료 처리'에서 다룹니다)
    }

    // 월 누적
    if (month === currentMonth) {
      monthlyProfit += profit;
      if (monthlyFirstCapital === null && !isNaN(capital)) {
        monthlyFirstCapital = capital;   // ★ 달의 첫 행에서만 세팅
      }
    }

    // 다음 행의 달 확인 (innerText -> textContent로 변경)
    const nextRow = rows[i + 1];
    let nextMonth = null;
    if (nextRow) {
        const nextDateCell = nextRow.querySelector(".date-cell");
        // textContent를 사용하고 trim()으로 혹시 모를 공백 제거
        nextMonth = nextDateCell ? nextDateCell.textContent.trim().slice(0, 7) : null;
    }

    // ▶ 달 종료 처리: 소계 행 추가
    if (month !== nextMonth) {
      const baseCapital = (typeof monthlyFirstCapital === 'number') ? monthlyFirstCapital : 0;
      const rate = (baseCapital > 0) ? (monthlyProfit / baseCapital * 100).toFixed(2) : "-";

      const usdToKrw = 1511.07;      
      const monthlyProfitKRW = usdToKrw ? Math.round(monthlyProfit * usdToKrw).toLocaleString('ko-KR') : "0";

      const subtotalRow = document.createElement("tr");
      subtotalRow.className = "subtotal-row table-success table-bordered text-end fw-bold";
      subtotalRow.innerHTML = `
        <td colspan="2" class="text-dark text-center">${currentMonth} 소계</td>
        <td class="text-end text-dark subtotal-capital">$${(baseCapital || 0).toFixed(0)}</td>
        <td class="text-end subtotal-profit">
          <div>$${monthlyProfit.toFixed(2)}</div>
          <div class="small">(${monthlyProfitKRW}원)</div>
        </td>
        <td class="text-end subtotal-rate">${rate}%</td>
        <td></td>
      `;
      row.insertAdjacentElement("afterend", subtotalRow);

      // 소계 색상 처리
      const scProfitCell = subtotalRow.querySelector(".subtotal-profit");
      const scRateCell   = subtotalRow.querySelector(".subtotal-rate");
      const profitValue  = monthlyProfit;
      const rateValue    = parseFloat(rate);

      if (scProfitCell) {
        scProfitCell.style.color = (profitValue < 0) ? '#0000CC' : (profitValue > 0 ? '#CC0000' : '');
      }
      if (scRateCell) {
        if (!isNaN(rateValue)) {
          scRateCell.style.color = (rateValue < 0) ? '#0000CC' : (rateValue > 0 ? '#CC0000' : '');
        }
      }


      // ▶ 전체 합계용 누적 (소계 기준)
      subtotalProfitSum += monthlyProfit;
      subtotalCapitalSum += (baseCapital || 0);
      subtotalCount++;

      // 다음 달로 초기화
      currentMonth = nextMonth;
      monthlyProfit = 0;
      monthlyFirstCapital = null;
    }
  }

  // (선택) 전체 합계 행이 필요하면 아래 주석 해제해서 사용
  /*
  const avgTotalCapital = subtotalCount > 0 ? (subtotalCapitalSum / subtotalCount) : 0;
  const totalRate = avgTotalCapital > 0 ? (subtotalProfitSum / avgTotalCapital * 100).toFixed(2) : "-";

  const totalRow = document.createElement("tr");
  totalRow.className = "total-row table-light text-end fw-bold";
  totalRow.innerHTML = `
    <td colspan="2" class="text-center">시드(월 첫값) 평균 | 수익합계</td>
    <td class="text-end">$${avgTotalCapital.toFixed(0)}</td>
    <td class="text-end total-profit">$${subtotalProfitSum.toFixed(2)}</td>
    <td class="text-end total-rate">${totalRate}%</td>
    <td></td>
  `;
  table.appendChild(totalRow);

  // 합계 색상
  const totalProfitCell = totalRow.querySelector(".total-profit");
  const totalRateCell   = totalRow.querySelector(".total-rate");
  if (subtotalProfitSum > 0) totalProfitCell.style.color = '#ff7f7f';
  else if (subtotalProfitSum < 0) totalProfitCell.style.color = '#3498db';

  if (totalRate !== '-' && parseFloat(totalRate) >= 0) totalRateCell.style.color = '#ff7f7f';
  else if (totalRate !== '-' && parseFloat(totalRate) < 0) totalRateCell.style.color = '#3498db';
  */

  // --- 여기부터 합계 계산 추가 ---
  let totalProfit = 0;

  rows.forEach(row => {
    const profitText = row.querySelector(".profit-cell")?.innerText.replace(/[^\d.\-]/g, '');
    const profit = parseFloat(profitText || 0);
    totalProfit += isNaN(profit) ? 0 : profit;
  });

  const summaryEl = document.getElementById("totalProfitSummary");
  if (summaryEl) {
    const EPS = 1e-9;
    if (Math.abs(totalProfit) < EPS) {
      summaryEl.style.display = "none";      // 0이면 숨김
    } else {
      const usdToKrw = 1511.07;
      const profitUSD = totalProfit.toFixed(2);
      const profitKRW = usdToKrw ? Math.round(totalProfit * usdToKrw).toLocaleString('ko-KR') : "0";

      summaryEl.textContent = `총 수익금 : $${profitUSD} (${profitKRW}원)`;

      const color = (totalProfit < 0) ? "#3498db" : "#ff7f7f";
      summaryEl.style.setProperty("color", color, "important");

      summaryEl.style.display = "";           // 보이기
    }
  }
}

<script>
(function () {
  const $cb = document.getElementById('auto_recommend');
  const $status = document.getElementById('autoRecommendStatus');
  if (!$cb) return;

  const getCsrf = () => {
    const m = document.querySelector('meta[name="csrf-token"]');
    return m ? m.getAttribute('content') : null;
  };

  const updateStatus = (on) => {
    if (!$status) return;
    $status.textContent = on ? "[ ON ]" : "[ OFF ]";
    $status.classList.toggle("text-success", on);
    $status.classList.toggle("text-secondary", !on);
  };

  $cb.addEventListener('change', async (e) => {
    const url = e.target.dataset.url;
    if (!url) return;

    const next = e.target.checked;
    e.target.disabled = true;

    try {
      const res = await fetch(url, {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          ...(getCsrf() ? {'X-CSRFToken': getCsrf()} : {})
        },
        body: JSON.stringify({ enabled: next })
      });
      const data = await res.json().catch(() => ({}));

      if (!res.ok || !data.ok) {
        e.target.checked = !next;
        updateStatus(!next);
        const msg = (data && data.message) ? data.message : '저장 실패';
        alert(msg);
      } else {
        updateStatus(next);
        alert('적용되었습니다');        
      }
    } catch (err) {
      e.target.checked = !next;
      updateStatus(!next);
      if (window.toast) toast('통신 오류로 저장하지 못했습니다.', 'danger');
      else alert('통신 오류로 저장하지 못했습니다.');
    } finally {
      e.target.disabled = false;
    }
  });
})();


function toggleAutoRecord(accountId) {

  const checkbox = document.getElementById('autoRecordSwitch');
  const isTurningOn = checkbox.checked; // 현재 체크된 상태 (ON을 시도하는 경우)
  const autoCompoundingCheckbox = document.getElementById('autoCompoundingSwitch');
  const isAutoCompoundingOn = autoCompoundingCheckbox && autoCompoundingCheckbox.checked;
  
  // 1. ON 할 때만 조건 체크
  if (isTurningOn) {
      const startDateStr = checkbox.dataset.startDate; // "2023-10-27" 형태
      const today = new Date();
      
      // 날짜 포맷팅 (YYYY-MM-DD 형식으로 맞춤)
      const year = today.getFullYear();
      const month = String(today.getMonth() + 1).padStart(2, '0');
      const day = String(today.getDate()).padStart(2, '0');
      const todayStr = `${year}-${month}-${day}`;

      if (startDateStr !== todayStr) {
          alert("사이클 시작일이 오늘인 경우에만 활성화할 수 있습니다. (새 사이클 시작 시)");
          checkbox.checked = false; // 체크박스 해제
          return;
      }
  } else {
      // OFF 할 때, 자동 재투자가 ON이면 함께 OFF시키기
      if (isAutoCompoundingOn) {
          if (!confirm("자동 기록 모드를 OFF하면 자동 재투자도 함께 OFF됩니다.\n계속하시겠습니까?")) {
              // 취소 시 체크박스 원복
              checkbox.checked = !checkbox.checked;
              return;
          }
          // 자동 재투자도 OFF로 변경 (API 호출)
          toggleAutoCompoundingOff(accountId);
          return;
      }
  }

  if (!confirm("기록 모드를 변경하시겠습니까?")) {
      // 취소 시 체크박스 원복
      document.getElementById('autoRecordSwitch').checked = !document.getElementById('autoRecordSwitch').checked;
      return;
  }

  fetch(`/api/account/${accountId}/toggle-auto-record`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' }
  })
  .then(res => res.json())
  .then(data => {
      if (data.success) {
          // 모드 변경 후 UI 동기화를 위해 페이지 새로고침
          location.reload();
      } else {
          alert("설정 변경 중 오류가 발생했습니다.");
      }
  })
  .catch(err => console.error("Error:", err));
}

// 자동 재투자를 OFF하는 헬퍼 함수
function toggleAutoCompoundingOff(accountId) {
    const autoCompoundingCheckbox = document.getElementById('autoCompoundingSwitch');
    autoCompoundingCheckbox.checked = false;

    fetch(`/api/account/${accountId}/toggle-auto-compounding`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' }
    })
    .then(res => res.json())
    .then(data => {
        if (data.success) {
            // 자동 재투자 OFF 후 자동 기록 모드 OFF 진행
            toggleAutoRecordContinue(accountId);
        } else {
            alert("자동 재투자 설정 변경 중 오류가 발생했습니다.");
            autoCompoundingCheckbox.checked = true;
        }
    })
    .catch(err => {
        console.error("Error:", err);
        autoCompoundingCheckbox.checked = true;
    });
}

// 자동 기록 모드를 계속 OFF하는 헬퍼 함수
function toggleAutoRecordContinue(accountId) {
    fetch(`/api/account/${accountId}/toggle-auto-record`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' }
    })
    .then(res => res.json())
    .then(data => {
        if (data.success) {
            alert("자동 재투자가 함께 OFF되었습니다.\n(자동 기록 모드는 OFF로 변경됩니다.)");
            location.reload();
        } else {
            alert("설정 변경 중 오류가 발생했습니다.");
            document.getElementById('autoRecordSwitch').checked = true;
        }
    })
    .catch(err => {
        console.error("Error:", err);
        document.getElementById('autoRecordSwitch').checked = true;
    });
}

function toggleAutoCompounding(accountId) {
    const checkbox = document.getElementById('autoCompoundingSwitch');
    const isTurningOn = checkbox.checked;
    
    // ✅ [추가] 자동 기록 모드 체크 (문자열 "true"와 비교)
    const isAutoRecordOn = checkbox.dataset.autoRecord === 'true';

    if (isTurningOn && !isAutoRecordOn) {
        alert("자동 재투자 기능을 켜려면 먼저 '자동 기록 모드'를 활성화해야 합니다.");
        checkbox.checked = false; // 스위치 끄기
        return;
    }

    // 변경 확인 컨펌
    const msg = isTurningOn 
        ? "자동 재투자를 활성화하시겠습니까?\n(손익 발생 시 다음 사이클 시드에 자동으로 적용됩니다.)" 
        : "자동 재투자를 비활성화하시겠습니까?";

    if (!confirm(msg)) {
        checkbox.checked = !isTurningOn;
        return;
    }

    // API 요청
    fetch(`/api/account/${accountId}/toggle-auto-compounding`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' }
    })
    .then(res => res.json())
    .then(data => {
        if (data.success) {
            location.reload();
        } else {
            alert("설정 변경 중 오류가 발생했습니다.");
            checkbox.checked = !isTurningOn;
        }
    })
    .catch(err => {
        console.error("Error:", err);
        checkbox.checked = !isTurningOn;
    });
}
